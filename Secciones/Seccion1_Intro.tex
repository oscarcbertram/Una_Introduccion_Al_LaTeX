\section{Introducción}
\subsection{¿Qué es esto del \LaTeX?}

\begin{frame}[t]{Un procesador de textos: Del \TeX{} al \LaTeX{}}%[plain] para que ocupe toda la pantalla
\begin{block}{El \TeX{}}
\begin{itemize}
\item \pause Creado por Donald E. Knuth (Univ. de Stanford)
\item \pause Proviene de la palabra {\em technology}, cuya raíz griega es $\tau\epsilon\chi\equiv$ {\em arte}
\item \pause Como todo lenguaje de programación:
\begin{itemize}
\item Gran {\color{miverdeO}potencia} y {\color{miverdeO}flexibilidad}
\item {\color{red}Complejidad} para aquellos no familiarizados con la programación
\end{itemize}
\end{itemize}
\end{block}

\begin{block}{El \LaTeX}
\begin{itemize}
\item \pause Leslie Lamport ({\em Digital Equipment Corp.})
\item \pause Conjunto de {\bf órdenes} que acercan el \TeX{} a los mortales
\item \pause Se normaliza el \LaTeX{}: \LaTeX$2_\varepsilon$
\end{itemize}
\end{block}
\end{frame}

\subsection{¿Por qué utilizarlo?}

\begin{frame}{¿Por qué utilizarlo?}
\begin{multicols}{2}
\begin{itemize}
\item \pause Es software libre
\item \pause Funciona sobre casi cualquier tipo de sistema y arquitectura (incluso las gratuitas)
\item \pause {\color{red}Auto}enumera las fórmulas y ecuaciones
\item \pause Crea índices de contenido, tablas, figuras y de términos ¡{\color{red}auto}máticamente!
\item \pause Permite el uso de Bib\TeX{}
\item \pause Resultado profesional
\item \pause Buen soporte para las matemáticas
\item \pause Para que las cosas queden monas ya están otros: {\color{red}\LaTeX{} lo hace por tí}
\item \pause Los archivos ocupan muy poco
\end{itemize}
\end{multicols}
\end{frame}
