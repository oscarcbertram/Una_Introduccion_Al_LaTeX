\begin{frame}{Introducción}

\begin{frame}[fragile]{Según la familia}
\begin{itemize}
\item <1-> Básicamente, \LaTeX{} tiene tres familias de letras:
\begin{itemize}
\item <2-> \alert<2>{roman (normal), \textbf{por defecto}}

\item <4-> \alert<4>{typewriter (máquina de escribir)}
\end{itemize}
\end{itemize}

\texttt{\backS texttt\{}{\em Texto}\texttt{\}} o bien \texttt{\{\backS ttfamily }{\em Texto}\texttt{\}} o bien \texttt{\{\backS tt }{\em Texto}\texttt{\}}\\

\subsection{Paginación de un documento}

\begin{frame}{Estilos de página}
\begin{itemize}
\item <1->Los estilos de página determinan el contenido del encabezamiento y del pie de página
\item <2-> Existen varios estilos de página, pero los más importantes son:
\only<3->{
\begin{description}
\item [\texttt{\color{black}plain}:] Cabercera vacía y pie con el número de página centrado 
\item [\texttt{\color{black}empty}:] Cabecera y pie vacíos
\item [\texttt{\color{black}headings}:] Cabecera con el número de página y un texto y el pie está vacío
\end{description}
}
\end{itemize}

\only<4->{Para elegir el estilo se emplean las siguientes declaraciones:}
\only<5->{
\begin{sintax}{Estilo de página}
\texttt{\backS pagestyle\{}{\em estilo}\texttt{\}} (hasta que aparezca otro comando)

 o bien

\texttt{\backS thispagestyle\{}{\em estilo}\texttt{\}} (para una sola página)
\end{sintax}
}

\only<6->{
\begin{itemize}
\item <6->\texttt{\backS pagestyle} lo emplearemos en el preámbulo para dar estilo a todo el documento
\item <7-> \texttt{\backS thispagestyle} tras \texttt{\backS maketitle} evita la numeración de la portada.
\end{itemize}
}

\end{frame}

\begin{frame}[fragile]{Parámetros más importantes de una página}
\only<1-6>{
\begin{itemize}
\item <1-6>El layout de una página se puede modificar al gusto (i.e.: para aprovechar más el papel)
\item <2-6> La {\only<3>{\color{blue}}mancha}\only<3>{ ¿\neutro?} del texto se puede modificar \only<3-4>{en:}
\only<4>{
\begin{itemize}
\item Tamaño (altura y anchura)
\item Posición (márgenes horizontal y vertical)
\end{itemize}}
\item <5-6>Los parámetros que modificaremos son:
\end{itemize}
}
\only<6->{
\begin{sintax}{Parámetros}
\texttt{\backS textwidth} {\em valor} \dotfill(anchura de la mancha)\\
\texttt{\backS textheight} {\em valor} \dotfill(altura de la mancha)\\
\texttt{\backS hoffset} {\em valor}\dotfill (márgen de salida horizontal)\\
\texttt{\backS voffset} {\em valor} \dotfill(márgen de salida vertical)\\
\texttt{\backS evensidemargin}  {\em valor}\dotfill(distancia de \texttt{\backS hoffset} a la mancha en pares)\\
\texttt{\backS oddsidemargin} {\em valor}\dotfill (distancia de \texttt{\backS hoffset} a la mancha en impares)\\
\end{sintax}
}

\subsection{Referencias cruzadas}
\begin{frame}[fragile]{Referencias}
\only<1->{
